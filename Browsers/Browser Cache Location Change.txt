# On Windows, works with admin rights only

## Chrome: Delete the "Cache" folder first

# PowerShell method:
New-Item -ItemType SymbolicLink -Path "$env:LocalAppData\Google\Chrome\User Data\Default\Cache" -Target "E:\Portables\Browser Cache\Chrome-Cache"

New-Item -ItemType SymbolicLink -Path "$env:LocalAppData\Google\Chrome\User Data\Default\Cache" -Target "E:\Portables\Browser Cache\Chrome-Cache"

# CMD method:
mklink /D "%LOCALAPPDATA%\Google\Chrome\User Data\Default\Cache" "E:\Portables\Browser Cache\Chrome-Cache"

# RamDisk Method

# PowerShell method:
New-Item -ItemType SymbolicLink -Path "$env:LocalAppData\Google\Chrome\User Data\Default\Cache" -Target "Z:\"

# CMD method:
mklink /D "%LOCALAPPDATA%\Google\Chrome\User Data\Default\Cache" "Z:\"

ln -s "/mnt/FaTbOy/Portables/Browser Cache/Edge-Linux" "$HOME/.cache/microsoft-edge/Default/Cache"
